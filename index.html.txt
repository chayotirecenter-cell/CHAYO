        /* NAVEGACIÓN Y MODO OSCURO */
        function switchTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(el => el.classList.remove('active'));
            document.querySelectorAll('.nav-btn').forEach(el => el.classList.remove('active'));
            
            document.getElementById(tabId).classList.add('active');
            event.currentTarget.classList.add('active');

            if(tabId === 'inventoryTab') renderInventory();
            if(tabId === 'reportsTab') renderReports();
        }

        function toggleTheme() {
            const body = document.body;
            const icon = document.querySelector('#themeToggle i');
            if (body.getAttribute('data-theme') === 'dark') {
                body.removeAttribute('data-theme');
                icon.className = 'fa-solid fa-moon';
            } else {
                body.setAttribute('data-theme', 'dark');
                icon.className = 'fa-solid fa-sun';
            }
        }

        /* PESTAÑA 1: SISTEMA CAJA / POS */
        function renderPosCatalog() {
            const search = document.getElementById('posSearch').value.toLowerCase();
            const topGrid = document.getElementById('topProductsGrid');
            const catalogGrid = document.getElementById('catalogGrid');

            topGrid.innerHTML = '';
            catalogGrid.innerHTML = '';

            // Top Vendidos (Ordenado por salesCount)
            const sortedItems = [...db.items].sort((a, b) => b.salesCount - a.salesCount).slice(0, 4);
            sortedItems.forEach(item => {
                topGrid.appendChild(createProductCard(item));
            });

            // Catálogo Filtrado
            const filteredItems = db.items.filter(i => i.name.toLowerCase().includes(search));
            filteredItems.forEach(item => {
                catalogGrid.appendChild(createProductCard(item));
            });
        }

        function createProductCard(item) {
            const card = document.createElement('div');
            card.className = 'product-card';
            card.onclick = () => addToCart(item.id);
            const badgeClass = item.type === 'Producto' ? 'badge-prod' : 'badge-serv';
            
            card.innerHTML = `
                <span class="product-badge ${badgeClass}">${item.type}</span>
                <div class="product-name">${item.name}</div>
                <div class="product-price">S/ ${item.price.toFixed(2)}</div>
                <div style="font-size:0.75rem; color:var(--text-muted);">${item.type === 'Producto' ? 'Stock: '+item.stock : 'Servicio'}</div>
            `;
            return card;
        }

        function addToCart(itemId) {
            const item = db.items.find(i => i.id === itemId);
            if (!item) return;

            if (item.type === 'Producto' && item.stock <= 0) {
                alert('¡Atención! Este producto no tiene stock suficiente.');
                return;
            }

            const existing = currentCart.find(c => c.id === itemId);
            if (existing) {
                if (item.type === 'Producto' && existing.qty >= item.stock) {
                    alert('Límite de stock alcanzado en carrito.');
                    return;
                }
                existing.qty++;
            } else {
                currentCart.push({ ...item, qty: 1 });
            }
            renderCart();
        }

        function updateCartQty(id, delta) {
            const index = currentCart.findIndex(c => c.id === id);
            if (index > -1) {
                currentCart[index].qty += delta;
                if (currentCart[index].qty <= 0) {
                    currentCart.splice(index, 1);
                }
            }
            renderCart();
        }

        function renderCart() {
            const container = document.getElementById('cartItems');
            container.innerHTML = '';

            if (currentCart.length === 0) {
                container.innerHTML = `<p style="text-align: center; color: var(--text-muted); padding: 20px;">Sin ítems en el carrito</p>`;
                updateCartTotals();
                return;
            }

            currentCart.forEach(item => {
                const row = document.createElement('div');
                row.className = 'cart-item';
                row.innerHTML = `
                    <div class="cart-item-details">
                        <div class="cart-item-title">${item.name}</div>
                        <div class="cart-item-sub">S/ ${item.price.toFixed(2)} x ${item.qty}</div>
                    </div>
                    <div style="display:flex; align-items:center; gap:5px;">
                        <button class="btn-qty" onclick="updateCartQty(${item.id}, -1)">-</button>
                        <span>${item.qty}</span>
                        <button class="btn-qty" onclick="updateCartQty(${item.id}, 1)">+</button>
                    </div>
                `;
                container.appendChild(row);
            });

            updateCartTotals();
        }

        function updateCartTotals(changedSource) {
            const subtotal = currentCart.reduce((sum, item) => sum + (item.price * item.qty), 0);
            let discSoles = parseFloat(document.getElementById('discountSoles').value) || 0;
            let discPercent = parseFloat(document.getElementById('discountPercent').value) || 0;

            if (changedSource === 'soles') {
                discPercent = subtotal > 0 ? (discSoles / subtotal) * 100 : 0;
                document.getElementById('discountPercent').value = discPercent.toFixed(1);
            } else if (changedSource === 'percent') {
                discSoles = (subtotal * discPercent) / 100;
                document.getElementById('discountSoles').value = discSoles.toFixed(2);
            }

            const total = Math.max(0, subtotal - discSoles);

            document.getElementById('cartSubtotal').innerText = `S/ ${subtotal.toFixed(2)}`;
            document.getElementById('cartDiscountLabel').innerText = `- S/ ${discSoles.toFixed(2)} (${discPercent.toFixed(0)}%)`;
            document.getElementById('cartTotal').innerText = `S/ ${total.toFixed(2)}`;
        }

        function completeSale() {
            if (currentCart.length === 0) {
                alert('Añade al menos un producto o servicio antes de cobrar.');
                return;
            }

            const client = document.getElementById('clientName').value.trim() || 'Cliente General';
            const plate = document.getElementById('carPlate').value.trim().toUpperCase() || 'S/N';
            const payment = document.getElementById('paymentMethod').value;
            const subtotal = currentCart.reduce((sum, item) => sum + (item.price * item.qty), 0);
            const discSoles = parseFloat(document.getElementById('discountSoles').value) || 0;
            const total = Math.max(0, subtotal - discSoles);

            // Registrar transacción
            const saleRecord = {
                id: Date.now(),
                date: new Date().toISOString(),
                client,
                plate,
                paymentMethod: payment,
                items: [...currentCart],
                subtotal,
                discount: discSoles,
                total
            };

            db.sales.push(saleRecord);

            // Actualizar Inventario y Contadores
            currentCart.forEach(cartItem => {
                const invItem = db.items.find(i => i.id === cartItem.id);
                if (invItem) {
                    if (invItem.type === 'Producto') {
                        invItem.stock -= cartItem.qty;
                    }
                    invItem.salesCount += cartItem.qty;
                }
            });

            saveToStorage();

            // Limpieza
            currentCart = [];
            document.getElementById('clientName').value = '';
            document.getElementById('carPlate').value = '';
            document.getElementById('discountSoles').value = 0;
            document.getElementById('discountPercent').value = 0;
            renderCart();
            renderPosCatalog();

            alert('¡Venta realizada con éxito!');
        }

        /* PESTAÑA 2: INVENTARIO */
        function renderInventory() {
            const tbody = document.getElementById('inventoryTableBody');
            tbody.innerHTML = '';

            db.items.forEach(item => {
                const tr = document.createElement('tr');
                tr.innerHTML = `
                    <td><span class="product-badge ${item.type==='Producto'?'badge-prod':'badge-serv'}">${item.type}</span></td>
                    <td><strong>${item.name}</strong></td>
                    <td>S/ ${item.price.toFixed(2)}</td>
                    <td>${item.type === 'Producto' ? item.stock : 'N/A'}</td>
                    <td>${item.unit || '-'}</td>
                    <td>${item.salesCount}</td>
                    <td>
                        <button class="btn-secondary" onclick="editItem(${item.id})"><i class="fa-solid fa-pen"></i></button>
                        <button class="btn-secondary" style="background-color:var(--primary-red);" onclick="deleteItem(${item.id})"><i class="fa-solid fa-trash"></i></button>
                    </td>
                `;
                tbody.appendChild(tr);
            });
        }

        function saveItem(e) {
            e.preventDefault();
            const id = document.getElementById('itemId').value;
            const type = document.getElementById('itemType').value;
            const name = document.getElementById('itemName').value;
            const price = parseFloat(document.getElementById('itemPrice').value);
            const stock = parseInt(document.getElementById('itemStock').value) || 0;
            const unit = document.getElementById('itemUnit').value;

            if (id) {
                // Editar
                const item = db.items.find(i => i.id == id);
                if (item) {
                    item.type = type;
                    item.name = name;
                    item.price = price;
                    item.stock = stock;
                    item.unit = unit;
                }
            } else {
                // Nuevo
                db.items.push({
                    id: Date.now(),
                    type,
                    name,
                    price,
                    stock,
                    unit,
                    salesCount: 0
                });
            }

            saveToStorage();
            resetItemForm();
            renderInventory();
            renderPosCatalog();
        }

        function editItem(id) {
            const item = db.items.find(i => i.id === id);
            if (!item) return;

            document.getElementById('itemId').value = item.id;
            document.getElementById('itemType').value = item.type;
            document.getElementById('itemName').value = item.name;
            document.getElementById('itemPrice').value = item.price;
            document.getElementById('itemStock').value = item.stock;
            document.getElementById('itemUnit').value = item.unit || '';
        }

        function deleteItem(id) {
            if (confirm('¿Deseas eliminar este registro?')) {
                db.items = db.items.filter(i => i.id !== id);
                saveToStorage();
                renderInventory();
                renderPosCatalog();
            }
        }

        function resetItemForm() {
            document.getElementById('itemForm').reset();
            document.getElementById('itemId').value = '';
        }

        /* BACKUP E IMPORTACIÓN DE DATOS */
        function exportData() {
            const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(db));
            const dlAnchorElem = document.createElement('a');
            dlAnchorElem.setAttribute("href", dataStr);
            dlAnchorElem.setAttribute("download", `chayo_backup_${new Date().toISOString().slice(0,10)}.json`);
            dlAnchorElem.click();
        }

        function importData(event) {
            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const importedDb = JSON.parse(e.target.result);
                    if (importedDb.items && importedDb.sales) {
                        db = importedDb;
                        saveToStorage();
                        alert('¡Base de datos restaurada correctamente!');
                        renderInventory();
                        renderPosCatalog();
                    } else {
                        alert('El archivo no es válido.');
                    }
                } catch(err) {
                    alert('Error al leer el archivo JSON.');
                }
            };
            reader.readAsText(event.target.files[0]);
        }

        /* PESTAÑA 3: REPORTES & INGRESOS */
        function setReportFilter(filter) {
            activeReportFilter = filter;
            document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
            if(filter === 'week') document.getElementById('btnFilterWeek').classList.add('active');
            if(filter === 'month') document.getElementById('btnFilterMonth').classList.add('active');
            if(filter === 'year') document.getElementById('btnFilterYear').classList.add('active');
            
            renderReports();
        }

        function renderReports() {
            const now = new Date();
            let filteredSales = db.sales.filter(s => {
                const sDate = new Date(s.date);
                if (activeReportFilter === 'week') {
                    const diffTime = Math.abs(now - sDate);
                    const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
                    return diffDays <= 7;
                } else if (activeReportFilter === 'month') {
                    return sDate.getMonth() === now.getMonth() && sDate.getFullYear() === now.getFullYear();
                } else if (activeReportFilter === 'year') {
                    return sDate.getFullYear() === now.getFullYear();
                }
                return true;
            });

            // Totales por Productos vs Servicios
            let prodTotal = 0;
            let servTotal = 0;

            filteredSales.forEach(sale => {
                sale.items.forEach(item => {
                    const lineTotal = item.price * item.qty;
                    if (item.type === 'Producto') prodTotal += lineTotal;
                    else servTotal += lineTotal;
                });
            });

            document.getElementById('kpiProducts').innerText = `S/ ${prodTotal.toFixed(2)}`;
            document.getElementById('kpiServices').innerText = `S/ ${servTotal.toFixed(2)}`;
            document.getElementById('kpiTotal').innerText = `S/ ${(prodTotal + servTotal).toFixed(2)}`;

            // Tabla Historial
            const historyBody = document.getElementById('salesHistoryBody');
            historyBody.innerHTML = '';
            filteredSales.sort((a,b) => new Date(b.date) - new Date(a.date)).forEach(s => {
                const tr = document.createElement('tr');
                const d = new Date(s.date);
                const dateStr = `${d.toLocaleDateString()} ${d.toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})}`;
                
                // Determinar si es mixto, solo prod o solo serv
                const hasProd = s.items.some(i => i.type === 'Producto');
                const hasServ = s.items.some(i => i.type === 'Servicio');
                let typeBadge = 'Mixto';
                if (hasProd && !hasServ) typeBadge = 'Producto';
                if (!hasProd && hasServ) typeBadge = 'Servicio';

                tr.innerHTML = `
                    <td>${dateStr}</td>
                    <td><strong>${s.client}</strong> <br><small style="color:var(--text-muted)">Placa: ${s.plate}</small></td>
                    <td>${s.paymentMethod}</td>
                    <td><span class="product-badge ${typeBadge==='Producto'?'badge-prod':(typeBadge==='Servicio'?'badge-serv':'')}">${typeBadge}</span></td>
                    <td>S/ ${s.discount.toFixed(2)}</td>
                    <td><strong>S/ ${s.total.toFixed(2)}</strong></td>
                `;
                historyBody.appendChild(tr);
            });

            // Generar Gráfico Chart.js
            renderChart(filteredSales);
        }
